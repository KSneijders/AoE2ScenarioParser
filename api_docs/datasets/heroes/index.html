            HeroInfo.POLEMARCH,
        ]
        heroes = []
        heroes.extend(base)
        if include_chronicles:
            heroes.extend(chronicles)
        return heroes

def trainable_heroes(
    include_chronicles: bool = False,
) -> List[HeroInfo]:
    """
    Args:
        include_chronicles: if set to `True`, include chronicles heroes

    Returns:
        A list of all trainable heroes
    """
    base = [
        HeroInfo.LIU_BEI,
        HeroInfo.CAO_CAO,
        HeroInfo.SUN_JIAN,
    ]
    chronicles = [
        HeroInfo.POLEMARCH,
    ]
    heroes = []
    heroes.extend(base)
    if include_chronicles:
        heroes.extend(chronicles)
    return heroes
