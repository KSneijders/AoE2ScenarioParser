120            rotation: float = 0,
121            garrisoned_in_id: int = -1,
122            animation_frame: int = 0,
123            status: int = 2,
124            reference_id: int = None,
125            caption_string_id: int = -1,
126            caption_string: str = '',
127            tile: Tile | Tuple[int, int] = None,
128    ) -> Unit:
129        """
130        Adds a unit to the scenario.
131
132        Args:
133            player: The player the unit belongs to.
134            unit_const: Defines what unit you're placing. The IDs used in the unit/buildings dataset.
135            x: The x location in the scenario.
136            y: The y location in the scenario.
137            z: The z (height) location in the scenario.
138            rotation: The rotation of the unit.
139            garrisoned_in_id: The reference_id of another unit this unit is garrisoned in.
140            animation_frame: The animation frame of the unit.
141            status: Unknown - Always 2. 0-6 no difference (?) | 7-255 makes it disappear. (Except from the mini-map)
142            reference_id: The reference ID of this unit. Normally added automatically. Used for garrisoning or reference
143                in triggers
144            caption_string_id: A string ID for the caption of a unit (mut. exclusive to caption_string)
145            caption_string: A string for the caption of a unit (mut. exclusive to caption_string_id)
146            tile: An object that represents a tile on the map. Replaces parameters x and y. Also, automatically adds
147                .5 to both ints to place the unit centered on the tile.
148
149        Returns:
150            The Unit created
151        """
152        if reference_id is None:
153            reference_id = self.get_new_reference_id()
154
155        caption_string_id_retriever = Unit._link_list[1].group[9]
156        if not caption_string_id_retriever.support.supports(self.get_scenario().scenario_version):
157            caption_string_id = None
158
159        caption_string_retriever = Unit._link_list[1].group[10]
160        if not caption_string_retriever.support.supports(self.get_scenario().scenario_version):
161            caption_string = None
162
163        unit = Unit(
164            player=player,
165            x=x if tile is None else (tile[0] + .5),
166            y=y if tile is None else (tile[1] + .5),
167            z=z,
168            reference_id=reference_id,
169            unit_const=unit_const,
170            status=status,
171            rotation=rotation,
172            initial_animation_frame=animation_frame,
173            garrisoned_in_id=garrisoned_in_id,
174            caption_string_id=caption_string_id,
175            caption_string=caption_string,
176            uuid=self._uuid
177        )
178
179        self.units[player].append(unit)
180        return unit

199    def filter_units_by(
200            self,
201            attr: str,
202            unit_attrs: List[int],
203            blacklist: bool = False,
204            player_list: List[Union[int, PlayerId]] = None,
205            unit_list: List[Unit] = None
206    ) -> List[Unit]:
207        """
208        Filter units based on a given attribute of units
209
210        Args:
211            attr: The attribute to filter by
212            unit_attrs: The values for the attributes to filter with
213            blacklist: Use the given constant list as blacklist instead of whitelist
214            player_list: A list of players to filter from. If not used, all players are used.
215            unit_list: A set of units to filter from. If not used, all units are used.
216
217        Returns:
218            A list of units
219
220        Raises:
221            AttributeError: If the provided attr does not exist on objects of the Unit class
222        """
223
224        if unit_list is None:
225            unit_list = self.get_all_units()
226        if player_list is not None:
227            unit_list = [unit for unit in unit_list if unit.player in player_list]
228
229        if len(unit_list) == 0:
230            return []
231
232        unit = unit_list[0]
233        if not hasattr(unit, attr):
234            raise AttributeError(f"Cannot filter Unit objects by {attr}")
235
236        # Both return statements can be combined using: ((unit.unit_const in unit_consts) != blacklist)
237        # But splitting them helps performance (not checking against blacklist for each entry)
238        if not blacklist:
239            return [unit for unit_attr in unit_attrs for unit in unit_list if getattr(unit, attr) == unit_attr]
240        return [unit for unit_attr in unit_attrs for unit in unit_list if getattr(unit, attr) != unit_attr]

284    def get_units_in_area(
285            self,
286            x1: float = None,
287            y1: float = None,
288            x2: float = None,
289            y2: float = None,
290            tile1: Tile = None,
291            tile2: Tile = None,
292            unit_list: List[Unit] = None,
293            players: List[Union[int, PlayerId]] = None,
294            ignore_players: List[PlayerId] = None
295    ) -> List[Unit]:
296        """
297        Returns all units in the square with left corner (x1, y1) and right corner (x2, y2). Both corners inclusive.
298
299        Args:
300            x1: The X location of the left corner
301            y1: The Y location of the left corner
302            x2: The X location of the right corner
303            y2: The Y location of the right corner
304            tile1: The x,y location of the 1st corner as Tile Object
305            tile2: The x,y location of the 2nd corner as Tile Object
306            unit_list: (Optional) A list of units (Defaults to all units in the map, including GAIA (Trees etc.)
307            players: (Optional) A list of Players which units need to be selected from the selected area
308            ignore_players: (Optional) A list of Players which units need to be ignored from the selected area
309
310        Raises:
311            ValueError: if not all 4 (x1, y1, x2 and y2) are used simultaneously.
312                Or if both (tile1 and tile2) are not used simultaneously.
313                Or if any of the 4 (x1, y1, x2, y2) is used together with any of (tile1, tile2). Use one or the other.
314                Or if players and ignore_players are used simultaneously.
315        """
316        if (x1 is not None or y1 is not None or x2 is not None or y2 is not None) and any([tile1, tile2]):
317            raise ValueError("Cannot use both x1,y1,x2,y2 notation and tile1,tile2 notation at the same time")
318        if (x1 is not None or y1 is not None or x2 is not None or y2 is not None) and \
319                (x1 is None or y1 is None or x2 is None or y2 is None):
320            raise ValueError("Cannot use some but not all from x1,y1,x2,y2.")
321        if (not all([tile1, tile2])) and any([tile1, tile2]):
322            raise ValueError("Cannot use one from tile1, tile2. Use both.")
323        if players is not None and ignore_players is not None:
324            raise ValueError("Cannot use both whitelist (players) and blacklist (ignore_players) at the same time")
325
326        if tile1:
327            x1 = tile1.x
328            y1 = tile1.y
329            x2 = tile2.x
330            y2 = tile2.y
331        else:
332            # Inclusive selection
333            x2 += 1
334            y2 += 1
335
336        if players is not None:
337            players = players
338        elif ignore_players is not None:
339            players = [p for p in PlayerId if p not in ignore_players]
340        else:
341            players = [p for p in PlayerId]
342
343        if unit_list is None:
344            unit_list = self.get_all_units()
345
346        return [unit for unit in unit_list if x1 <= unit.x <= x2 and y1 <= unit.y <= y2 and unit.player in players]

372    def find_highest_reference_id(self) -> int:
373        """
374        Find the highest ID in the map. Searches through all units for the highest ID.
375
376        Returns:
377            The highest ID in the map
378        """
379        highest_id = 0  # If no units, default to 0
380        for player in PlayerId.all():
381            for unit in self.units[player]:
382                highest_id = max(highest_id, unit.reference_id)
383        return highest_id

385    def remove_unit(self, reference_id: int = None, unit: Unit = None) -> None:
386        """
387        Removes a unit. Please note that `unit=...` is a lot faster than `reference_id=...` due to reference_id having
388        to search through all units on the map. And unit has an ownership (player) attribute which is used for knowing
389        which list to remove the unit from.
390
391        Args:
392            reference_id: The id of the unit. Note that this is NOT a unit constant (So NOT: UnitInfo.ARCHER)
393            unit: The Unit object to be removed.
394        """
395        if reference_id is not None and unit is not None:
396            raise ValueError("Cannot use both unit_ref_id and unit arguments. Use one or the other.")
397        if reference_id is None and unit is None:
398            raise ValueError("Both unit_ref_id and unit arguments were unused. Use one.")
399
400        if reference_id is not None:
401            for player in range(0, 9):
402                for i, unit in enumerate(self.units[player]):
403                    if unit.reference_id == reference_id:
404                        del self.units[player][i]
405                        return
406        elif unit is not None:
407            self.units[unit.player].remove(unit)

297        """
298        Returns all units in the square with left corner (x1, y1) and right corner (x2, y2). Both corners inclusive.
299
300        Args:
301            x1: The X location of the left corner
302            y1: The Y location of the left corner
303            x2: The X location of the right corner
304            y2: The Y location of the right corner

306            tile2: The x,y location of the 2nd corner as Tile Object
307            unit_list: (Optional) A list of units (Defaults to all units in the map, including GAIA (Trees etc.)
308            players: (Optional) A list of Players which units need to be selected from the selected area
309            ignore_players: (Optional) A list of Players which units need to be ignored from the selected area
310
311        Raises:
312            ValueError: if not all 4 (x1, y1, x2 and y2) are used simultaneously.
313                Or if both (tile1 and tile2) are not used simultaneously.
314                Or if any of the 4 (x1, y1, x2, y2) is used together with any of (tile1, tile2). Use one or the other.
315                Or if players and ignore_players are used simultaneously.
316        """
317        if (x1 is not None or y1 is not None or x2 is not None or y2 is not None) and any([tile1, tile2]):
318            raise ValueError("Cannot use both x1,y1,x2,y2 notation and tile1,tile2 notation at the same time")
319        if (x1 is not None or y1 is not None or x2 is not None or y2 is not None) and \
320                (x1 is None or y1 is None or x2 is None or y2 is None):
321            raise ValueError("Cannot use some but not all from x1,y1,x2,y2.")
322        if (not all([tile1, tile2])) and any([tile1, tile2]):
323            raise ValueError("Cannot use one from tile1, tile2. Use both.")
324        if players is not None and ignore_players is not None:
325            raise ValueError("Cannot use both whitelist (players) and blacklist (ignore_players) at the same time")
326
327        if tile1:
328            x1 = tile1.x
329            y1 = tile1.y
330            x2 = tile2.x
331            y2 = tile2.y
332        else:
333            # Inclusive selection
334            x2 += 1
335            y2 += 1
336
337        if players is not None:
338            players = players
339        elif ignore_players is not None:
340            players = [p for p in PlayerId if p not in ignore_players]
341        else:
342            players = [p for p in PlayerId]
343
344        if unit_list is None:
345            unit_list = self.get_all_units()
346
347        return [unit for unit in unit_list if x1 <= unit.x <= x2 and y1 <= unit.y <= y2 and unit.player in players]
