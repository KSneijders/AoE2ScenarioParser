 72        """
 73        Object to easily select an area on the map. Uses method chaining for ease of use.
 74
 75        **Please note**: Setting a ``uuid`` will always overwrite the ``map_size`` attribute, even if it's not ``None``.
 76
 77        Args:
 78            map_size: The size of the map this area object will handle
 79            uuid: The UUID of the scenario this area belongs to
 80            x1: The X location of the left corner
 81            y1: The Y location of the left corner
 82            x2: The X location of the right corner
 83            y2: The Y location of the right corner
 84            corner1: The location of the left corner
 85            corner2: The location of the right corner
 86        """
 87        if map_size is None and uuid is None:
 88            if corner1 is None and (x1 is None or y1 is None):
 89                raise ValueError("Cannot create area object without knowing the map size or a UUID from a scenario.")
 90
 91        self.uuid: UUID = uuid
 92        if map_size is not None:
 93            self._map_size_value = map_size
 94        else:
 95            self._map_size_value = None
 96
 97        if values_are_valid(x1, y1) or value_is_valid(corner1):
 98            x1, y1, x2, y2 = validate_coords(x1, y1, x2, y2, corner1, corner2)
 99        else:

244    def to_coords(self, as_terrain: bool = False) -> OrderedSet[Tile | 'TerrainTile']:
245        """
246        Converts the selection to an OrderedSet of (x, y) coordinates
247
248        Args:
249            as_terrain: If the returning coordinates should be Tile objects or Terrain Tiles. If `True` the coordinates
250                are returned as TerrainTiles.
251
252        Returns:
253            An OrderedSet of Tiles ((x, y) named tuple) of the selection.
254
255        Examples:
256            The selection: ``((3,3), (5,5))`` would result in an OrderedSet with a length of 9::
257
258                [
259                    (3,3), (4,3)  ...,
260                    ...,   ...,   ...,
261                    ...,   (4,5), (5,5)
262                ]
263        """
264        tiles = OrderedSet(
265            Tile(x, y) for y in self.get_range_y() for x in self.get_range_x() if self.is_within_selection(x, y)
266        )
267        return self._tiles_to_terrain_tiles(tiles) if as_terrain else tiles

269    def to_chunks(
270            self,
271            as_terrain: bool = False
272    ) -> List[OrderedSet[Tile | 'TerrainTile']]:
273        """
274        Converts the selection to a list of OrderedSets with Tile NamedTuples with (x, y) coordinates.
275        The separation between chunks is based on if they're connected to each other.
276        So the tiles must share an edge (i.e. they should be non-diagonal).
277        (With exceptions like grid state when using gap_size = 0)
278
279        Args:
280            as_terrain: If the returning coordinates should be Tile objects or Terrain Tiles. If `True` the coordinates
281                are returned as TerrainTiles.
282
283        Returns:
284            A list of OrderedSets with Tiles ((x, y) named tuple) of the selection.
285        """
286        tiles = self.to_coords()
287
288        # Shortcut for states that CANNOT be more than one chunk
289        if self.state in AreaState.unchunkables():
290            return [tiles]
291
292        chunks: Dict[int, List[Tile]] = {}
293        for tile in tiles:
294            chunk_id = self._get_chunk_id(tile)
295            chunks.setdefault(chunk_id, []).append(tile)
296
297        map_size = self._map_size
298        chunks_ordered: List[OrderedSet[Tile | 'TerrainTile']] = []
299        for chunk_id, chunk_tiles in chunks.items():
300            tiles = self._tiles_to_terrain_tiles(chunk_tiles) if as_terrain else chunk_tiles
301            chunks_ordered.append(
302                OrderedSet(sorted(tiles, key=lambda t: t.y * map_size + t.x))
303            )
304
305        return chunks_ordered

307    def to_chunk_areas(self) -> List[Area]:
308        """
309        Converts the selection to a list of Area objects.
310        Internally, this uses `to_chunks` to convert the selection to chunks.
311        The separation between chunks is based on if they're connected to each other.
312        So the tiles must share an edge (i.e. they should be non-diagonal).
313        (With exceptions like grid state when using gap_size = 0)
314
315        Returns:
316            A list with Area objects based on the selection.
317        """
318        return [Area(corner1=chunk[0], corner2=chunk[-1]) for chunk in self.to_chunks()]

533    def move_to(self, corner: Literal['west', 'north', 'east', 'south'], x: int, y: int):
534        """
535        Moves the selection area to the given coordinate by placing the given corner of this area on the coordinate.
536
537        For center placement, use ``.center(...)``
538        """
539        width = self.get_width() - 1
540        height = self.get_height() - 1
541
542        if corner == 'west':
543            self.x1, self.y1, self.x2, self.y2 = x, y, x + width, y + height
544        elif corner == 'north':
545            self.x1, self.y1, self.x2, self.y2 = x - width, y, x, y + height
546        elif corner == 'east':
547            self.x1, self.y1, self.x2, self.y2 = x - width, y - height, x, y
548        elif corner == 'south':
549            self.x1, self.y1, self.x2, self.y2 = x, y - height, x + width, y
550
551        return self

553    def size(self, n: int) -> Area:
554        """
555        Sets the selection to a size around the center. If center is (4,4) with a size of 3 the selection will become
556        ``((3,3), (5,5))``
557        """
558        center_x, center_y = self.get_center_int()
559        n -= 1  # Ignore center tile
560        self.x1 = center_x - math.ceil(n / 2)
561        self.y1 = center_y - math.ceil(n / 2)
562        self.x2 = center_x + math.floor(n / 2)
563        self.y2 = center_y + math.floor(n / 2)
564        return self

605        """
606        Moves the selection center to a given position on the map. This function makes sure it cannot go over the edge
607        of the map. The selection will be forced against the edge of the map and the selection will not be decreased in
608        size.
609        """
610        self._force_map_size()
611
612        center_x, center_y = self.get_center()

614        if diff_x < 0 and abs(diff_x) > self.x1:
615            diff_x = -self.x1
616        elif diff_x > 0 and diff_x > (distance_x := self.maximum_coordinate - self.x2):
617            diff_x = distance_x
618        if diff_y < 0 and abs(diff_y) > self.y1:
619            diff_y = -self.y1
620        elif diff_y > 0 and diff_y > (distance_y := self.maximum_coordinate - self.y2):
621            diff_y = distance_y
622        self.x1 += diff_x
623        self.y1 += diff_y
624        self.x2 += diff_x
625        self.y2 += diff_y
626        return self

773    def _get_length_change(self, new_len: int, cur_len: int, first_coord: int, second_coord) -> Tuple[int, int]:
774        """
775        Calculate the differences in tiles for the 2 points (x1 & x2) or (y1 & y2) when the length of an edge is changed
776
777        Args:
778            new_len: The new length
779            cur_len: The current length
780            first_coord: Coord of the first corner (x1 or y1)
781            second_coord: Coord of the first corner (x2 or y2)
782
783        Returns:
784            The differences for the first and second coordinate. Can be negative and positive ints.
785        """
786        half: float = (new_len - cur_len) / 2
787        half1, half2 = -half, half
788        if half > 0:
789            if half > first_coord:
790                half1 = -first_coord
791                half2 += half - first_coord
792            if half > (dist := self.maximum_coordinate - second_coord):
793                half2 = dist
794                half1 += half - dist
795            return math.floor(half1), math.floor(half2)
796        return math.ceil(half1), math.ceil(half2)

853    def _get_chunk_id(self, tile: Tile) -> int:
854        """
855        This function gets the Chunk id of a tile based on the current state and configs. The chunk ID identifies which
856        chunk the given tile is in. This is useful for separating chunks that are connected but shouldn't be in the same
857        chunk (like when creating a checker or stripe pattern)
858
859        Args:
860            tile: The tile to check as Tile object
861
862        Returns:
863            The int ID of the chunk, or, -1 when it's not in a selection, or 0 when the selection cannot be split into
864                chunks.
865
866        Raises:
867            ValueError: if the area configuration isn't supported by this function.
868        """
869        if not self.is_within_selection(tile=tile):
870            return -1
871
872        if self.state in AreaState.unchunkables():
873            return 0
874
875        elif self.state == AreaState.GRID:
876            if self.inverted:
877                return 0
878            per_row = math.ceil(self.get_width() / (self.block_size_x + self.gap_size_x))
879            return (tile.x - self.x1) // (self.block_size_x + self.gap_size_x) + \
880                (tile.y - self.y1) // (self.block_size_y + self.gap_size_y) * per_row
881
882        elif self.state == AreaState.LINES:
883            if self.axis == "x":
884                return (tile.y - self.y1) // (self.line_width_y + self.gap_size_y)
885            elif self.axis == "y":
886                return (tile.x - self.x1) // (self.line_width_x + self.gap_size_x)
887
888        elif self.state == AreaState.CORNERS:
889            # 0 Left, 1 Top, 2 Right, 3 Bottom
890            if self.x1 <= tile.x < self.x1 + self.corner_size_x and self.y1 <= tile.y < self.y1 + self.corner_size_y:
891                return 0
892            if self.x2 - self.corner_size_x < tile.x <= self.x2 and self.y1 <= tile.y < self.y1 + self.corner_size_y:
893                return 1
894            if self.x2 - self.corner_size_x < tile.x <= self.x2 and self.y2 - self.corner_size_y < tile.y <= self.y2:
895                return 2
896            if self.x1 <= tile.x < self.x1 + self.corner_size_x and self.y2 - self.corner_size_y < tile.y <= self.y2:
897                return 3
898        raise ValueError(f"Invalid area configuration for getting the Chunk ID. If you believe this is an error, "
899                         f"please raise an issue on github or in the Discord server")

254            An OrderedSet of Tiles ((x, y) named tuple) of the selection.
255
256        Examples:
257            The selection: ``((3,3), (5,5))`` would result in an OrderedSet with a length of 9::
258
259                [
260                    (3,3), (4,3)  ...,
261                    ...,   ...,   ...,
262                    ...,   (4,5), (5,5)
263                ]
264        """
265        tiles = OrderedSet(
266            Tile(x, y) for y in self.get_range_y() for x in self.get_range_x() if self.is_within_selection(x, y)
267        )
268        return self._tiles_to_terrain_tiles(tiles) if as_terrain else tiles
269
270    def to_chunks(
271            self,
272            as_terrain: bool = False
273    ) -> List[OrderedSet[Tile | 'TerrainTile']]:
274        """
275        Converts the selection to a list of OrderedSets with Tile NamedTuples with (x, y) coordinates.
276        The separation between chunks is based on if they're connected to each other.
277        So the tiles must share an edge (i.e. they should be non-diagonal).
278        (With exceptions like grid state when using gap_size = 0)
279
280        Args:
281            as_terrain: If the returning coordinates should be Tile objects or Terrain Tiles. If `True` the coordinates
282                are returned as TerrainTiles.
283
284        Returns:
285            A list of OrderedSets with Tiles ((x, y) named tuple) of the selection.
286        """
287        tiles = self.to_coords()
288
289        # Shortcut for states that CANNOT be more than one chunk
290        if self.state in AreaState.unchunkables():
291            return [tiles]
292
293        chunks: Dict[int, List[Tile]] = {}
294        for tile in tiles:
295            chunk_id = self._get_chunk_id(tile)
296            chunks.setdefault(chunk_id, []).append(tile)
297
298        map_size = self._map_size
299        chunks_ordered: List[OrderedSet[Tile | 'TerrainTile']] = []
300        for chunk_id, chunk_tiles in chunks.items():
301            tiles = self._tiles_to_terrain_tiles(chunk_tiles) if as_terrain else chunk_tiles
302            chunks_ordered.append(
303                OrderedSet(sorted(tiles, key=lambda t: t.y * map_size + t.x))
304            )
305
306        return chunks_ordered
307
308    def to_chunk_areas(self) -> List[Area]:
309        """
310        Converts the selection to a list of Area objects.
311        Internally, this uses `to_chunks` to convert the selection to chunks.
312        The separation between chunks is based on if they're connected to each other.
313        So the tiles must share an edge (i.e. they should be non-diagonal).
314        (With exceptions like grid state when using gap_size = 0)
315
316        Returns:
317            A list with Area objects based on the selection.
318        """
319        return [Area(corner1=chunk[0], corner2=chunk[-1]) for chunk in self.to_chunks()]

534    def move_to(self, corner: Literal['west', 'north', 'east', 'south'], x: int, y: int):
535        """
536        Moves the selection area to the given coordinate by placing the given corner of this area on the coordinate.
537
538        For center placement, use ``.center(...)``
539        """
540        width = self.get_width() - 1
541        height = self.get_height() - 1
542
543        if corner == 'west':
544            self.x1, self.y1, self.x2, self.y2 = x, y, x + width, y + height
545        elif corner == 'north':
546            self.x1, self.y1, self.x2, self.y2 = x - width, y, x, y + height
547        elif corner == 'east':
548            self.x1, self.y1, self.x2, self.y2 = x - width, y - height, x, y
549        elif corner == 'south':
550            self.x1, self.y1, self.x2, self.y2 = x, y - height, x + width, y

589    def center(self, x: int, y: int) -> Area:
590        """
591        Moves the selection center to a given position. When the given center forces the selection of the edge of the
592        map the off-map tiles will not be returned. When moving the selection back into the map the tiles will be
593        returned again.
594
595        If you want to limit moving the center without changing the selection box size, use: ``center_bounded``

598        diff_x, diff_y = math.floor(x - center_x), math.floor(y - center_y)
599        self.x1 = self._x1 + diff_x

601        self.x2 = self._x2 + diff_x
602        self.y2 = self._y2 + diff_y

605    def center_bounded(self, x: int, y: int) -> Area:
606        """
607        Moves the selection center to a given position on the map. This function makes sure it cannot go over the edge
608        of the map. The selection will be forced against the edge of the map and the selection will not be decreased in
609        size.
610        """
611        self._force_map_size()
612
613        center_x, center_y = self.get_center()
614        diff_x, diff_y = math.floor(x - center_x), math.floor(y - center_y)
615        if diff_x < 0 and abs(diff_x) > self.x1:
616            diff_x = -self.x1
617        elif diff_x > 0 and diff_x > (distance_x := self.maximum_coordinate - self.x2):
618            diff_x = distance_x
619        if diff_y < 0 and abs(diff_y) > self.y1:
620            diff_y = -self.y1
621        elif diff_y > 0 and diff_y > (distance_y := self.maximum_coordinate - self.y2):
622            diff_y = distance_y
623        self.x1 += diff_x
624        self.y1 += diff_y
625        self.x2 += diff_x
626        self.y2 += diff_y
627        return self

711    def is_within_selection(self, x: int = -1, y: int = -1, tile: Tile = None) -> bool:
712        """
713        If a given (x,y) location is within the selection.
714
715        Args:
716            x: The X coordinate
717            y: The Y coordinate
718            tile: A Tile object, replacing the x & y coordinates
719
720        Returns:
721            `True` if (x,y) is within the selection, `False` otherwise
722        """
723        if tile is not None:
724            x, y = tile
725
726        if not (self._x1 <= x <= self._x2 and self._y1 <= y <= self._y2):
727            return False
728
729        is_within: bool
730        if self.state == AreaState.EDGE:
731            is_within = self._is_edge_tile(x, y)
732        elif self.state == AreaState.GRID:
733            is_within = self._is_a_grid_tile(x, y)
734        elif self.state == AreaState.LINES:
735            is_within = self._is_a_line_tile(x, y)
736        elif self.state == AreaState.CORNERS:
737            is_within = self._is_a_corner_tile(x, y)
738        else:
739            is_within = True
740        return self._invert_if_inverted(is_within)
741
742    def is_within_bounds(self) -> bool:
743        """Check if the current selection is within the map"""
744        self._force_map_size()
745
746        return 0 <= self._x1 < self.map_size \
747            and 0 <= self._y1 < self.map_size \
748            and 0 <= self._x2 < self.map_size \
749            and 0 <= self._y2 < self.map_size

774    def _get_length_change(self, new_len: int, cur_len: int, first_coord: int, second_coord) -> Tuple[int, int]:
775        """
776        Calculate the differences in tiles for the 2 points (x1 & x2) or (y1 & y2) when the length of an edge is changed
777
778        Args:
779            new_len: The new length
780            cur_len: The current length
781            first_coord: Coord of the first corner (x1 or y1)
782            second_coord: Coord of the first corner (x2 or y2)
783
784        Returns:
785            The differences for the first and second coordinate. Can be negative and positive ints.
786        """
787        half: float = (new_len - cur_len) / 2
788        half1, half2 = -half, half
789        if half > 0:
790            if half > first_coord:
791                half1 = -first_coord
792                half2 += half - first_coord
793            if half > (dist := self.maximum_coordinate - second_coord):
794                half2 = dist
795                half1 += half - dist
796            return math.floor(half1), math.floor(half2)
797        return math.ceil(half1), math.ceil(half2)

799    def _is_edge_tile(self, x: int, y: int) -> bool:
800        """ Returns if a given tile (x,y) is an edge tile of the set selection given a certain edge width."""
801        return any((
802            0 <= x - self.x1 < self.line_width_x,
803            0 <= y - self.y1 < self.line_width_y,
804            0 <= self.x2 - x < self.line_width_x,
805            0 <= self.y2 - y < self.line_width_y
806        ))

809        """Inverts the boolean if the area is in inverted state."""

812    def _is_a_corner_tile(self, x: int, y: int) -> bool:
813        """If a given (x,y) location is a corner tile."""
814        return ((self.x1 <= x < self.x1 + self.corner_size_x) or (self.x2 - self.corner_size_x < x <= self.x2)) and \
815            ((self.y1 <= y < self.y1 + self.corner_size_y) or (self.y2 - self.corner_size_y < y <= self.y2))
816
817    def _is_a_grid_tile(self, x: int, y: int) -> bool:
818        """If a given (x,y) location is within the grid selection."""
819        return (x - self.x1) % (self.block_size_x + self.gap_size_x) < self.block_size_x and \
820            (y - self.y1) % (self.block_size_y + self.gap_size_y) < self.block_size_y

822    def _is_a_line_tile(self, x: int, y: int) -> bool:
823        """If a given (x,y) location is within the grid selection."""
824        if self.axis == "x":
825            return (y - self.y1) % (self.gap_size_y + self.line_width_y) < self.line_width_y
826        elif self.axis == "y":
827            return (x - self.x1) % (self.gap_size_x + self.line_width_x) < self.line_width_x
828        raise ValueError("Invalid axis value. Should be either x or y")

831        """Keeps a given value within the bounds of ``0 <= val < map_size``."""

854    def _get_chunk_id(self, tile: Tile) -> int:
855        """
856        This function gets the Chunk id of a tile based on the current state and configs. The chunk ID identifies which
857        chunk the given tile is in. This is useful for separating chunks that are connected but shouldn't be in the same
858        chunk (like when creating a checker or stripe pattern)
859
860        Args:
861            tile: The tile to check as Tile object
862
863        Returns:
864            The int ID of the chunk, or, -1 when it's not in a selection, or 0 when the selection cannot be split into
865                chunks.
866
867        Raises:
868            ValueError: if the area configuration isn't supported by this function.
869        """
870        if not self.is_within_selection(tile=tile):
871            return -1
872
873        if self.state in AreaState.unchunkables():
874            return 0
875
876        elif self.state == AreaState.GRID:
877            if self.inverted:
878                return 0
879            per_row = math.ceil(self.get_width() / (self.block_size_x + self.gap_size_x))
880            return (tile.x - self.x1) // (self.block_size_x + self.gap_size_x) + \
881                (tile.y - self.y1) // (self.block_size_y + self.gap_size_y) * per_row
882
883        elif self.state == AreaState.LINES:
884            if self.axis == "x":
885                return (tile.y - self.y1) // (self.line_width_y + self.gap_size_y)
886            elif self.axis == "y":
887                return (tile.x - self.x1) // (self.line_width_x + self.gap_size_x)
888
889        elif self.state == AreaState.CORNERS:
890            # 0 Left, 1 Top, 2 Right, 3 Bottom
891            if self.x1 <= tile.x < self.x1 + self.corner_size_x and self.y1 <= tile.y < self.y1 + self.corner_size_y:
892                return 0
893            if self.x2 - self.corner_size_x < tile.x <= self.x2 and self.y1 <= tile.y < self.y1 + self.corner_size_y:
894                return 1
895            if self.x2 - self.corner_size_x < tile.x <= self.x2 and self.y2 - self.corner_size_y < tile.y <= self.y2:
896                return 2
897            if self.x1 <= tile.x < self.x1 + self.corner_size_x and self.y2 - self.corner_size_y < tile.y <= self.y2:
898                return 3
899        raise ValueError(f"Invalid area configuration for getting the Chunk ID. If you believe this is an error, "
900                         f"please raise an issue on github or in the Discord server")

def to_chunks( self, as_terrain: bool = False) -> "List[OrderedSet[Tile | 'TerrainTile']]": 
270    def to_chunks(
271            self,
272            as_terrain: bool = False
273    ) -> List[OrderedSet[Tile | 'TerrainTile']]:
274        """
275        Converts the selection to a list of OrderedSets with Tile NamedTuples with (x, y) coordinates.
276        The separation between chunks is based on if they're connected to each other.
277        So the tiles must share an edge (i.e. they should be non-diagonal).
278        (With exceptions like grid state when using gap_size = 0)
279
280        Args:
281            as_terrain: If the returning coordinates should be Tile objects or Terrain Tiles. If `True` the coordinates
282                are returned as TerrainTiles.
283
284        Returns:
285            A list of OrderedSets with Tiles ((x, y) named tuple) of the selection.
286        """
287        tiles = self.to_coords()
288
289        # Shortcut for states that CANNOT be more than one chunk
290        if self.state in AreaState.unchunkables():
291            return [tiles]
292
293        chunks: Dict[int, List[Tile]] = {}
294        for tile in tiles:
295            chunk_id = self._get_chunk_id(tile)
296            chunks.setdefault(chunk_id, []).append(tile)
297
298        map_size = self._map_size
299        chunks_ordered: List[OrderedSet[Tile | 'TerrainTile']] = []
300        for chunk_id, chunk_tiles in chunks.items():
301            tiles = self._tiles_to_terrain_tiles(chunk_tiles) if as_terrain else chunk_tiles
302            chunks_ordered.append(
303                OrderedSet(sorted(tiles, key=lambda t: t.y * map_size + t.x))
304            )
305
306        return chunks_ordered
