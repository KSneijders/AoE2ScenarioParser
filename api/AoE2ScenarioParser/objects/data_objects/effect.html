486    def get_content_as_string(self, include_effect_definition: bool = False) -> str:
487        """
488        Create a human-readable string showcasing all content of this effect.
489
490        This is also the function called when doing: `print(effect)`
491
492        Args:
493            include_effect_definition: If the effect meta-data should be added to the result
494
495        Returns:
496            The created string
497        """
498        if self.effect_type not in effects.attributes:  # Unknown effect
499            attributes_list = list(effects.empty_attributes.keys())
500        else:
501            attributes_list = effects.attributes[self.effect_type]
502
503        return_string = ""
504        for attribute in attributes_list:
505            if self._is_attribute_supported(attribute):
506                val = getattr(self, attribute, None)
507            else:
508                val = None
509
510            if not self._should_be_displayed(attribute, val):
511                continue
512
513            value_string = transform_effect_attr_value(self.effect_type, attribute, val, self._uuid)

531    def _split_aa_value(self, quantity: int) -> Tuple[int, int]:
532        """
533        A function to convert the initial quantity value to the quantity and armor/attack values.
534        Unfortunately this problem has to be solved in the object due to how specific this was implemented in DE.
535
536        Args:
537            quantity: the initial quantity value
538
539        Returns:
540            The one byte armor/attack class as int and one byte armor/attack quantity as int
541
542        ----
543
544        **Trigger Version 2.4**::
545
546            Quantity value: (3, 5)
547            00000000 00000000 00000011 000000101
548                                aaq      aac
549
550        Final 2/4 bytes are aaq (1 byte), and aac (1 byte). First 2 are unused. Max value of both is 255.
551
552        **Trigger Version 2.5**::
553
554            Quantity value: (3, 5)
555            00000000 00000011 00000000 000000101
556              aaq      aaq      aac      aac
557
558        The 4/4 bytes are aaq (2 bytes), and aac (2 bytes). All are used. Max value of both is 65535.
559
560        ----
561        """
562        trigger_version = getters.get_trigger_version(self._uuid)
563        if trigger_version >= 2.5:
564            return quantity >> 16, quantity & 65535
565        return quantity >> 8, quantity & 255

567    def _merge_aa_values(self, aa_class: int, aa_quantity: int) -> int:
568        """
569        A function to convert the quantity and armor/attack field to a quantity value.
570        Unfortunately this problem has to be solved in the object due to how specific this was implemented in DE.
571
572        Args:
573            aa_quantity: the armor quantity value
574            aa_class: the armor/attack value
575
576        Returns:
577            The one byte quantity and one byte armor/attack value
578        """
579        trigger_version = getters.get_trigger_version(self._uuid)
580        if trigger_version >= 2.5:
581            return aa_class * 65536 + aa_quantity
582
583        # Would use `aa_class << 8` - but apparently multiplication is faster
584        return aa_class * 256 + aa_quantity
