263    def _load_content_sections(self, raw_file_igenerator: IncrementalGenerator) -> None:
264        """
265        Reads and adds all the remaining file sections from the structure file to the sections dict of the
266        scenario.
267
268        The sections after the header are compressed and are first decompressed using the -zlib.MAX_WBITS algorithm.
269
270        Args:
271            raw_file_igenerator: The generator to read the file sections from
272        """
273        self._decompressed_file_data = _decompress_bytes(raw_file_igenerator.get_remaining_bytes())
274
275        data_igenerator = IncrementalGenerator(name='Scenario Data', file_content=self._decompressed_file_data)
276
277        for section_name in self.structure.keys():
278            if section_name == "FileHeader":
279                continue
280            try:
281                if section_name == SectionName.TRIGGERS.value:
282                    trigger_version = bytes_to_double(data_igenerator.get_bytes(8, update_progress=False))
283                    self._validate_latest_trigger_data_version(trigger_version)
284
285                self._create_and_load_section(section_name, data_igenerator)
286            except UnsupportedVersionError as e:
287                raise e
288            except Exception as e:
289                print(f"\n[{e.__class__.__name__}] AoE2Scenario.parse_file: \n\tSection: {section_name}\n")
290                self.write_error_file(trail_generator=data_igenerator)
291                raise e

392    def _write_from_structure(self, filename: str, skip_reconstruction: bool = False) -> None:
393        """
394        Writes the scenario to a new file with the given filename
395
396        Args:
397            filename: The location to write the file to
398            skip_reconstruction: If reconstruction should be skipped. If true, this will ignore all changes made
399                using the managers (For example all changes made using trigger_manager).
400
401        Raises:
402            ValueError: if the setting DISABLE_ERROR_ON_OVERWRITING_SOURCE is not disabled and the source filename is
403                the same as the filename being written to
404        """
405        if not skip_reconstruction:
406            self.commit()
407
408        s_print("File writing from structure started...", final=True, time=True, newline=True)
409        binary = _get_file_section_data(self.sections.get('FileHeader'))
410
411        binary_list_to_be_compressed = []
412        for file_part in self.sections.values():
413            if file_part.name == "FileHeader":
414                continue
415            binary_list_to_be_compressed.append(_get_file_section_data(file_part))
416
417        compressed = _compress_bytes(b''.join(binary_list_to_be_compressed))
418
419        with open(filename, 'wb') as f:
420            f.write(binary + compressed)
421
422        etime = round(time.time() - self._time_start, 2)
423        s_print("File writing finished successfully.", final=True, time=True)
424        s_print(f"File successfully written to: " + color_string(f"'{filename}'", "magenta"), final=True, time=True)
425        s_print(f"Execution time from scenario read: {etime}s", final=True, time=True)

478    def _validate_scenario_variant(self):
479        if self.variant is None:
480            self._warn_variant_unknown()
481            return
482
483        # If the header has been adjusted manually (solution before this functionality went live)
484        if self.sections["FileHeader"].unknown_value_2 != self.variant.value:
485            self.variant = self.sections["FileHeader"].unknown_value_2

487        if self.variant == ScenarioVariant.ROR and self.scenario_version_tuple < (1, 49):

491                "Upgrade the scenario by saving it in the in-game editor before converting it."

494        if self.variant not in [ScenarioVariant.AOE2, ScenarioVariant.ROR] and settings.SHOW_VARIANT_WARNINGS:
495            applicants = self.variant.applicants()
496            name = self.variant.name

555    def _debug_write_from_source(self, filename: str, datatype: str, write_bytes: bool = True) -> None:
556        """
557        Writes the decompressed scenario file as bytes or as hex text
558
559        Args:
560            filename: The filename to write to
561            datatype: these are flags that indicate which parts of the file to include in the output. 'd' for
562                decompressed file data, 'f' for the file, and 'h' for the header. Note: Only 'd' actually works at this
563                time
564            write_bytes: boolean to determine if the file needs to be written as bytes or hex text form
565        """
566        s_print("File writing from source started with attributes " + datatype + "...")
567        file = open(filename, "wb" if write_bytes else "w")
568        selected_parts = []
569        for t in datatype:
570            if t == "f":
571                selected_parts.append(self._file)
572            elif t == "h":
573                selected_parts.append(self._file_header)
574            elif t == "d":
575                selected_parts.append(self._decompressed_file_data)
576        parts = None
577        for part in selected_parts:
578            if parts is None:
579                parts = part
580                continue
581            parts += part
582        file.write(parts if write_bytes else create_textual_hex(parts.hex()))
583        file.close()
584        s_print("File writing finished successfully.")
