  9    """
 10    This enum class provides information about most of the heroes in the game. Information about the following
 11    properties of a hero is found in this class:
 12
 13     - Unit ID
 14     - Icon ID
 15     - Dead Unit ID
 16     - HotKey ID

 23    >>> InfoDatasetBase.from_icon_id()

 25    >>> InfoDatasetBase.gaia_only()
 26    >>> InfoDatasetBase.non_gaia()
 27
 28    **Examples**
 29
 30    >>> HeroInfo.WILLIAM_WALLACE.ID
 31    >>> 432
 32
 33    >>> HeroInfo.WILLIAM_WALLACE.ICON_ID

 36    >>> HeroInfo.WILLIAM_WALLACE.DEAD_ID
 37    >>> 433
 38
 39    >>> HeroInfo.WILLIAM_WALLACE.HOTKEY_ID
 40    >>> 16299
 41
 42    >>> HeroInfo.WILLIAM_WALLACE.IS_GAIA_ONLY
 43    >>> False
 44    """
 45
 46    @staticmethod
 47    def trainable_heroes(
 48        include_chronicles: bool = False,
 49    ) -> List[HeroInfo]:

 51        Args:
 52            include_chronicles: if set to `True`, include chronicles heroes
 53
 54        Returns:
 55            A list of all trainable heroes
 56        """
 57        base = [
 58            HeroInfo.LIU_BEI,
 59            HeroInfo.CAO_CAO,
 60            HeroInfo.SUN_JIAN,
 61        ]
 62        chronicles = [
 63            HeroInfo.POLEMARCH,
 64        ]
 65        heroes = []
 66        heroes.extend(base)
 67        if include_chronicles:
 68            heroes.extend(chronicles)
 69        return heroes
