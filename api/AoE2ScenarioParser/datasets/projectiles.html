 12class ProjectileInfo(Enum):
 13    """
 14    This enum class provides information about the projectiles in the game. Information about the following properties
 15    of a projectile is found in this class:
 16
 17     - Projectile ID
 18     - The units that use the projectile
 19
 20    **Methods**
 21    >>> ProjectileInfo.from_id()
 22    >>> ProjectileInfo.get_unit_projectile()
 23
 24    **Examples**
 25
 26    >>> ProjectileInfo.SCORPION_FIRE.ID
 27    >>> 628
 28
 29    >>> ProjectileInfo.SCORPION_FIRE.USED_BY
 30    >>> (542,)
 31    """
 32
 33    @property
 34    def ID(self) -> int:
 35        """
 36        Returns:
 37            The ID of the specified projectile unit
 38        """
 39        return self.value[0]
 40
 41    @property
 42    def USED_BY(self) -> tuple[int]:
 43        """
 44        Returns:
 45            A tuple of unit IDs that use the specified projectile unit
 46        """
 47        return self.value[1]
 48
 49    @classmethod
 50    def from_id(cls, projectile_id: int) -> ProjectileInfo:
 51        """
 52        Get the ProjectileInfo object from its ID
 53
 54        Args:
 55            projectile_id: The ID of the projectile to get the ProjectileInfo of
 56
 57        Returns:
 58            A ProjectileInfo object of the specified projectile ID
 59        """
 60        if type(projectile_id) is not int:
 61            raise TypeError(f"from_id expected int, got {type(projectile_id)}")
 62        if projectile_id == -1:
 63            raise ValueError("-1 is not a valid projectile ID")
 64
 65        for projectile in cls._member_map_.values():
 66            if projectile.value[0] == projectile_id:
 67                return projectile
 68
 69        raise KeyError(f"A projectile unit with ID '{projectile_id}' was not found in the dataset")
 70
 71    @staticmethod
 72    def get_unit_projectile(unit_const: int, has_chemistry: bool = False, secondary: bool = False) -> ProjectileInfo:
 73
 74        """
 75        Get the projectile object based on unit const
 76
 77        Args:
 78            unit_const: The unit to find the projectile of
 79            has_chemistry: If you want the fire (chemistry) version (if it exists)
 80            secondary: If you want the secondary projectile version (if it exists)
 81
 82        Returns:
 83            The ProjectileInfo corresponding to the given unit and params or None if nothing was found
 84        """
 85
 86        if type(unit_const) is not int:
 87            raise TypeError(f"unit const expected int, got {type(unit_const)}")
 88        if unit_const < 0:
 89            raise ValueError(f"{unit_const} is not a valid unit constant")
 90
 91        projectile = "secondary" if secondary else "primary"
 92        fire = "_fire" if has_chemistry else ""
 93        projectile_id = _unit_projectile_info.get(str(unit_const), {}).get(projectile + fire, -1)
 94
 95        return ProjectileInfo.from_id(projectile_id) if projectile_id != -1 else None
